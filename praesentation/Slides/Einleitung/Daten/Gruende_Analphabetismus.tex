\frame{\frametitle{Gründe des Analphabetismus}
	\subsection{Gründe des Analphabetismus}
	
		\begin{enumerate}
		
			\item Mangelnde Bildung
			
			
			\item Legasthenie
			\begin{itemize}
				\item Eine Störung der auditiven und visuellen Wahrnehmungsverarbeitung.
			\end{itemize}
			
			
			\item Dyslexie
			\begin{itemize}
				\item Wörter und/oder Texte, werden cognitiv nicht richtig verstanden.
			\end{itemize}
			
			
			\item Argrafie
			\begin{itemize}
				\item Wörter können nicht geschrieben werden, trotz normalen Intellekts und guter Handmotorik
			\end{itemize}
		\end{enumerate}
	
	}
