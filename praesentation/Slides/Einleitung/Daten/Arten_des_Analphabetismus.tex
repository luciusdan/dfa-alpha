\frame{\frametitle{Arten des Analphabetismus}
	\subsection{Arten des Analphabetismus}
	
		\begin{itemize}
    			\item primärer Analphabetismus
    				\begin{itemize}
    					\item Wenn man das Lesen und Schreiben nie gelernt hat.
    				\end{itemize}
    				
			\item sekundärer Analphabetismus				
    				\begin{itemize}
    					\item Wenn das Lesen und Schreiben wieder verlernt wurde.
    				\end{itemize}
				
			\item Semianalphabetismus
				\begin{itemize}
    					\item Wenn man lesen, aber nicht schreiben kann
    				\end{itemize}
    				
			\item funktionaler Analphabetismus
				\begin{itemize}
    					\item Wenn man einzelne Worte versteht, aber mit langen Texten und deren Zusammenhänge massive Schwierigkeiten hat.
    				\end{itemize}
		\end{itemize}
